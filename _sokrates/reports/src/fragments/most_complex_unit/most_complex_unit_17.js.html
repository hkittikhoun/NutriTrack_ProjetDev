export function mergeCartWithCatalogue(cartFoods = [], foods = []) {
  const cartOptions = (cartFoods || []).map((f) => ({
    FoodID: f.foodId,
    FoodDescription: f.name,
    _fromCart: true,
  }));
  return [
    ...cartOptions,
    ...foods.filter(
      (fd) => !cartOptions.some((c) => String(c.FoodID) === String(fd.FoodID))
    ),
  ];
}
