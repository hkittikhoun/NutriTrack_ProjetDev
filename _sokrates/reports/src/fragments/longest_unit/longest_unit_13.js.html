export function prefillFromCartToIngredients(cartFoods) {
  if (!cartFoods || cartFoods.length === 0) return null;
  return cartFoods.map((c, idx) => ({
    id: Date.now() + idx,
    foodId: String(c.foodId),
    ingredientName: c.name,
    quantity: c.cartQuantity,
    unit: "g",
    preparation: "",
  }));
}
