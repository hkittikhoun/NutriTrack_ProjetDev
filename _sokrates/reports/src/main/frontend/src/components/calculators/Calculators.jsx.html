"active" : "" }`} onClick={() => handleTabChange("calories")} > Calorie Calculator </button> <button className={`tab-button ${ activeCalculator === "nutrition" ? "active" : "" }`} onClick={() => handleTabChange("nutrition")} > Nutrition Calculator </button> </div> <div className="calculator-content"> {activeCalculator === "calories" && <CaloriesCalculator />} {activeCalculator === "nutrition" && <NutritionCalculator />} </div> </div> ); }
